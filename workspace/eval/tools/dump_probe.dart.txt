// [템플릿·.txt로 보존 — analyze 제외] FID 렌더 덤프 프로브 (표준 pump 진입점 규약·산출물 무관 고정 템플릿)
//
// 산출물 위젯 트리를 {type,text?,icon?,img?,tap?,children} JSON으로 덤프한다 → eval `dump_to_ir.dart`가
// layout-ir로 정규화 → `compare_layout.dart`가 시안(`dddart/scripts/extract_layout.dart`)과 대조.
//
// ── 표준 pump 진입점 규약(왜 이 파일이 고정인가) ──
// 산출물 `test/<bc>/.../_support.dart`가 `screenProbes`(implementation-test §7·코퍼스·양판 미러)를 노출한다:
//   `final Map<String, ScreenProbe> screenProbes` — role 문자열('list'·'detail'…)→그 화면을 대표 fixture로
//   펌프하고 루트 finder를 반환하는 함수(`typedef ScreenProbe = Future<Finder> Function(WidgetTester)`).
// view 클래스명·헬퍼명·fixture명·패키지명이 전부 그 맵 *값 안에* 갇히므로, 이 프로브는 `_support.dart` 하나만
// 상대 import해 맵을 순회한다 — 산출물별 import/pump 손질이 0이다(coordinator가 배선 추론 없이 일관 덤프).
//
// 사용: 이 파일을 산출물의 `_support.dart`와 같은 디렉터리에 `fid_dump_test.dart`로 복사하고
//   `flutter test <그 경로>` 실행. systemTemp에 `code-tree-<role>.json`을 화면 role마다 하나씩 생성.
import 'dart:convert';
import 'dart:io';

import 'package:flutter/material.dart';
import 'package:flutter_test/flutter_test.dart';

import '_support.dart';

Map<String, dynamic> _dump(Element root) {
  Map<String, dynamic> node(Element e) {
    final Widget w = e.widget;
    final Map<String, dynamic> m = <String, dynamic>{'type': w.runtimeType.toString()};
    if (w is Text) m['text'] = w.data ?? '';
    if (w is Icon) m['icon'] = w.icon?.toString() ?? '';
    if (w is Image) m['img'] = true;
    if (w is InkWell) m['tap'] = w.onTap != null;
    final List<Map<String, dynamic>> kids = <Map<String, dynamic>>[];
    e.visitChildren((Element c) => kids.add(node(c)));
    if (kids.isNotEmpty) m['children'] = kids;
    return m;
  }

  return node(root);
}

void main() {
  final String out = Directory.systemTemp.path;
  // screenProbes(_support.dart)를 순회 — role마다 펌프 후 루트를 덤프. 화면 추가는 산출물 쪽 맵에만.
  screenProbes.forEach((String role, ScreenProbe probe) {
    testWidgets('dump $role tree', (WidgetTester tester) async {
      final Finder root = await probe(tester);
      File('$out/code-tree-$role.json')
          .writeAsStringSync(jsonEncode(_dump(tester.element(root.first))));
    });
  });
}
