Map<String, dynamic> _dump(Element root) {
  Map<String, dynamic> node(Element e) {
    final Widget w = e.widget;
    final Map<String, dynamic> m = <String, dynamic>{'type': w.runtimeType.toString()};
    if (w is Text) m['text'] = w.data ?? '';
    if (w is Icon) m['icon'] = w.icon?.toString() ?? '';
    if (w is Image) m['img'] = true;
    if (w is InkWell) m['tap'] = w.onTap != null;
    final List<Map<String, dynamic>> kids = <Map<String, dynamic>>[];
    e.visitChildren((Element c) => kids.add(node(c)));
    if (kids.isNotEmpty) m['children'] = kids;
    return m;
  }

  return node(root);
}

void main() {
  final String out = Directory.systemTemp.path;
  testWidgets('dump weekly-list tree', (WidgetTester tester) async {
    final List<Forecast> week = <Forecast>[
      for (int i = 0; i < 7; i++)
        fc(d(i),
            condition: WeatherCondition.values[i % WeatherCondition.values.length],
            high: 28 - i,
            low: 19 - i),
    ];
    await pumpList(tester, week);
    File('$out/code-tree-weekly.json')
        .writeAsStringSync(jsonEncode(_dump(tester.element(find.byType(WeatherListView).first))));
  });

  testWidgets('dump daily-detail tree', (WidgetTester tester) async {
    await pumpDetail(tester, detailState(high: 28, low: 19));
    File('$out/code-tree-detail.json')
        .writeAsStringSync(jsonEncode(_dump(tester.element(find.byType(WeatherDetailView).first))));
  });
}
